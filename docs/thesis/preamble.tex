\usepackage{fp}

\def\makeTECS#1#2{%
  \foreach \index in {0, ..., #2} {%
    \FPset\x{#2}
    \FPeval\y{1.0/(x + 1.0) - 0.05}
    \begin{subfigure}[b]{\y\textwidth}
      \centering
      \inputtikz{#1_\index}
      \caption*{S[\index]}
    \end{subfigure}
  }}
