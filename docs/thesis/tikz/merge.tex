\begin{tikzpicture}[node distance = 1.2cm,
  on grid,
  auto,
]
\tikzstyle{every state}=[inner sep=0pt, minimum size=10pt, draw=none]

\node (u) [state] {$\textrm{u}$};
\node (n0) [state, below left = of u] {$\textrm{n}_{0}$};
\node (u1) [state, below right = of u] {$\textrm{u}_{1}$};
\node (n1) [state, below left = of u1] {$\textrm{n}_{1}$};
\node (dots) [state, below right = of u1] {\ldots};
\node (uk) [state, below right = of dots] {$\textrm{u}_{k-1}$};
\node (nk1) [state, below left = of uk] {$\textrm{n}_{k-1}$};
\node (nk) [state, below right = of uk] {$\textrm{n}_{k}$};

\path [-stealth, thick, color=black]
  (u) edge [dashed] node {} (n0)
  (u) edge [] node {} (u1)
  (u1) edge [dashed] node {} (n1)
  (u1) edge [] node {} (dots)
  (dots) edge [] node {} (uk)
  (uk) edge [dashed] node {} (nk1)
  (uk) edge [] node {} (nk)
  ;

\end{tikzpicture}
