\tikzstyle{every state}=[inner sep=0pt, minimum size=10pt, draw=none]

\node (u1) [state, below right = of u] {$\textrm{u}_{1}$};
\node (u2) [state, below right = of u1] {$\textrm{u}_{2}$};

\node (n2) [state, below = of u2] {};
\node (ln2) [state, below left = of n2] {l($\textrm{n}_{2}$)};

\node (ln1) [state, left = of rn1] {l($\textrm{n}_{1}$)};

\path [-stealth, thick, color=black]
  (u) edge [dashed, bend right] node {} (ln1)
  (u) edge [] node {} (u1)

  (u1) edge [dashed] node {} (ln2)
  (u1) edge [] node {} (u2)

  (u2) edge [dashed, bend left] node {} (rn2)
  (u2) edge [] node {} (rn1)
  ;
